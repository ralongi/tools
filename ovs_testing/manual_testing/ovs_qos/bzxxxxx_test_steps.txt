git_install() {
    if rpm -q git 2>/dev/null; then
        echo "Git is already installed; doing a git pull"; cd /mnt/tests/kernel; git pull
        return 0
    else
            yum -y install git
            mkdir /mnt/tests
        pushd /mnt/tests && git clone git://pkgs.devel.redhat.com/tests/kernel && popd
    fi
}
git_install
